import { Tab, Tabs } from "rspress/theme";

Add `live` when declaring a code block to turn it into a live demo.

<Tabs>
  <Tab label="JSX">
    This snippet:

    ````
    ```jsx live
    export const App = () => {
      return <div>Hello World</div>;
    };
    ```
    ````

    Turns into:

    ```jsx live
    export const App = () => {
      return <div>Hello World</div>;
    };
    ```

  </Tab>
	<Tab label="TSX">
    This snippet:

    ````
    ```tsx live
    import React from "react";

    export const App = (): React.FC => {
      return <div>Hello World</div>;
    };
    ```
    ````

    Turns into:

    ```tsx live
    import React from "react";

    export const App = (): React.FC => {
      return <div>Hello World</div>;
    };
    ```

  </Tab>
</Tabs>

> [!NOTE]
>
> - Component needs to be **exported**
> - External modules currently are not supported in inline demos
