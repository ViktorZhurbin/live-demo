import { Tab, Tabs } from "@rspress/core/theme";

By default, imports from `"@rspress/core/theme"` and `"react"` are available.

## Using built-in Rspress components

<Tabs>
  <Tab label="Live">
    ```jsx live
    import { Badge } from "@rspress/core/theme"

    export const App = () => {
      return (
        <Badge type="info">Some Badge</Badge>
      )
    }
    ```
  </Tab>
  <Tab label="file.mdx">
    <pre>
      ````
      ```jsx live
      import { Badge } from "@rspress/core/theme"

      export const App = () => {
        return (
          <Badge type="info">Some Badge</Badge>
        )
      }
      ```
      ````
    </pre>
  </Tab>
</Tabs>

## Using React hooks

<Tabs>
  <Tab label="Live">
    ```jsx live
    import React from "react"

    export const App = () => {
      const [count, setCount] = React.useState(0)

      const increment = () => {
        setCount(prev => prev + 1)
      }

      const button = (
        <button
          onClick={increment}
        >
          Increment
        </button>
      )

      return (
        <div>Count is: {count}</div>
      )
    }
    ```
  </Tab>
